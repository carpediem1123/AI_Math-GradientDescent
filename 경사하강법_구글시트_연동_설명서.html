STEP 1 — 구글 시트 + Apps Script 설정

약 5분 소요
구글 시트 새 파일을 만듭니다.
sheets.new 로 바로 접속하거나, 구글 드라이브에서 + 새로 만들기 → Google 스프레드시트를 선택하세요.
Apps Script를 엽니다.
상단 메뉴에서 확장 프로그램 → Apps Script 클릭
기존 코드를 모두 지우고 아래 코드를 붙여넣습니다.
편집창의 내용을 전체 선택(Ctrl+A) 후 삭제하고, 아래 코드를 붙여넣기(Ctrl+V)하세요.
Google Apps Script — 전체 코드 (복사해서 붙여넣기)
const SHEET_RAW    = '원시데이터';
const SHEET_RESULT = '결과요약';
const SHEET_SESPEC = '세특초안';

function doOptions(e) {
  return ContentService.createTextOutput('')
    .setMimeType(ContentService.MimeType.TEXT)
    .setHeaders({
      'Access-Control-Allow-Origin': '*',
      'Access-Control-Allow-Methods': 'POST',
      'Access-Control-Allow-Headers': 'Content-Type'
    });
}

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss   = SpreadsheetApp.getActiveSpreadsheet();
    const rawSheet    = getOrCreateSheet(ss, SHEET_RAW);
    const resultSheet = getOrCreateSheet(ss, SHEET_RESULT);
    const sespecSheet = getOrCreateSheet(ss, SHEET_SESPEC);
    initHeaders(rawSheet, resultSheet, sespecSheet);

    const ts = new Date().toLocaleString('ko-KR', {timeZone:'Asia/Seoul'});
    const isResend = data.type === 'resend';

    rawSheet.appendRow([
      ts, isResend ? '재전송' : '최초',
      data.cls, data.num, data.name,
      data.fa, data.da, data.a0, data.k, data.thresh,
      data.finalA, data.fillCount, data.fillDone,
      data.reflectDiscovery  || '',
      data.reflectApplication|| '',
      data.reflectCurious    || '',
      data.quizScore, data.quizTotal, (data.quizPct||0)+'%',
      data.q1||'—', data.q2||'—', data.q3||'—',
      data.q4||'—', data.q5||'—', data.q6||'—',
      data.seSpec || ''
    ]);

    upsertResult(resultSheet, ts, data, isResend);
    upsertSeSpec(sespecSheet, ts, data);

    return ContentService
      .createTextOutput(JSON.stringify({status:'ok', ts}))
      .setMimeType(ContentService.MimeType.JSON)
      .setHeader('Access-Control-Allow-Origin', '*');
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({status:'error', msg: err.message}))
      .setMimeType(ContentService.MimeType.JSON)
      .setHeader('Access-Control-Allow-Origin', '*');
  }
}

function getOrCreateSheet(ss, name) {
  return ss.getSheetByName(name) || ss.insertSheet(name);
}

function initHeaders(rawSheet, resultSheet, sespecSheet) {
  if (rawSheet.getLastRow() === 0) {
    rawSheet.appendRow([
      '전송시각','구분','반','번호','이름',
      'E(a)식','E\'(a)식','a₀','k','중단조건',
      '최종a','갱신횟수','탐구완료',
      '① 발견한 점','② AI 활용 생각','③ 추가 탐구',
      '퀴즈점수','총점','정답률',
      'Q1','Q2','Q3','Q4','Q5','Q6','세특초안'
    ]);
    rawSheet.getRange(1,1,1,rawSheet.getLastColumn())
      .setBackground('#1a1a2e').setFontColor('#ffffff').setFontWeight('bold');
    rawSheet.setFrozenRows(1);
  }
  if (resultSheet.getLastRow() === 0) {
    resultSheet.appendRow([
      '반','번호','이름','E(a)','E\'(a)','a₀','k','중단조건',
      '최종 a','갱신횟수','탐구완료','발견한 점 요약','AI 활용 연결',
      '퀴즈점수','총점','정답률(%)','성취수준','최초전송','마지막전송'
    ]);
    resultSheet.getRange(1,1,1,resultSheet.getLastColumn())
      .setBackground('#0e3d6e').setFontColor('#ffffff').setFontWeight('bold');
    resultSheet.setFrozenRows(1);
  }
  if (sespecSheet.getLastRow() === 0) {
    sespecSheet.appendRow(['반','번호','이름','성취수준','세특 초안','최근전송']);
    sespecSheet.getRange(1,1,1,6)
      .setBackground('#1a3a1a').setFontColor('#ffffff').setFontWeight('bold');
    sespecSheet.setColumnWidth(5, 700);
    sespecSheet.setFrozenRows(1);
  }
}

  let found = -1, firstTs = ts;

function colorLevel(sheet, row, col, level) {
  const bg ={'상':'#c6efce','중':'#ffeb9c','하':'#ffc7ce'};
  const fc ={'상':'#276221','중':'#9c6500','하':'#9c0006'};
  const cell=sheet.getRange(row,col);
  cell.setBackground(bg[level]||'#fff');
  cell.setFontColor(fc[level]||'#000');
  cell.setFontWeight('bold');
}

STEP 3 — HTML 파일에 URL 입력

파일 상단 입력칸에 한 번만 설정하면 됩니다
학생들이 접속한 HTML 파일을 교사 브라우저에서 엽니다.
(GitHub Pages 링크 또는 로컬 파일)
화면 상단 파란 배너에 Google Sheets URL 입력칸이 있습니다.
복사해 둔 웹앱 URL을 붙여넣고 [저장] 버튼을 클릭하세요.
✓ 저장됨 메시지가 나타나면 완료입니다.
브라우저 localStorage에 저장되므로 다음번 접속 시 자동으로 불러옵니다.
📌
URL은 교사 PC에만 저장되면 됩니다. 학생 PC에 따로 설정할 필요가 없습니다. 학생 화면에는 URL 입력칸이 보이지만, 비워두어도 전송에 필요한 URL은 파일 내부에 없으므로 학생이 임의로 변경할 수 없습니다.
단, GitHub Pages에 올린 파일은 코드를 수정해 URL을 하드코딩하는 방식이 더 안정적입니다. (아래 FAQ 참고)

자주 묻는 질문 (FAQ)

GitHub Pages에 올리면 URL이 매번 초기화되지 않나요? ▼
GitHub Pages에 올린 파일은 모든 학생이 같은 파일을 공유합니다. URL 입력칸에 입력해도 각자의 브라우저 localStorage에만 저장되므로, 교사 PC에서 설정해도 학생 PC에는 적용되지 않습니다.

해결 방법: HTML 파일의 id="gs-url" 입력칸의 placeholder 속성 옆에 value="https://script.google.com/..." 를 직접 넣어 저장한 뒤 GitHub에 업로드하면 됩니다.
코드 수정 후 재배포하면 URL이 바뀌나요? ▼
새 배포를 하면 URL이 바뀝니다. 기존 URL을 유지하려면 배포 → 배포 관리 → 기존 배포 선택 → 수정을 사용하세요. 이렇게 하면 같은 URL에서 업데이트된 코드로 작동합니다.
학생이 퀴즈를 다시 풀면 데이터가 덮어써지나요? ▼
원시데이터 탭에는 모든 전송 이력이 쌓입니다(최초/재전송 구분 표시).
결과요약·세특초안 탭은 같은 학생(반+번호+이름)이 재전송하면 최신 내용으로 업데이트됩니다. 최초 전송 시각은 보존됩니다.
학생이 이름을 잘못 입력하면 중복 행이 생기나요? ▼
반+번호+이름 세 가지를 키로 사용합니다. 이름이 조금이라도 다르면 별도 행으로 추가됩니다. 원시데이터 탭에서 확인 후 중복 행을 수동으로 삭제하시면 됩니다.
수업이 끝난 후 데이터 수집을 막으려면? ▼
Apps Script → 배포 → 배포 관리 → 해당 배포 선택 → 보관처리를 클릭하면 URL이 비활성화됩니다. 이후 전송 시도는 모두 오류로 처리됩니다.
태블릿(갤럭시탭, iPad)에서도 전송이 되나요? ▼
Safari(iOS)와 Chrome(Android) 모두 지원합니다. 단, Safari의 개인정보 보호 모드(Private)에서는 localStorage가 작동하지 않아 최초 전송 플래그가 저장되지 않을 수 있습니다. 일반 모드로 접속하도록 안내해 주세요.
